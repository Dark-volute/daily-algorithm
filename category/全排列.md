## 代码
```js

var permute = function(nums) {
  let res = []
  const used = new Array(nums.len);
  const dfs = (list) => {
    if(list.length === nums.length) {
      res.push(list.slice())
      return
    }
    for(let i = 0; i < nums.length; i++) {
      if (used[i]) {      // 这个数使用过了，跳过。
        continue;
       }
       list.push(nums[i])
        used[i] = true; 
        dfs(list)
        list.pop()
        used[i] = false; 
    }
  }
  dfs([])
  return res
};

```
