var permute = function(nums) {
  let res = []
  const dfs = (list) => {
    if(list.length === nums.length) {
      res.push(list.slice())
      return
    }
    for(let i = 0; i < nums.length; i++) {
      if (list.indexOf(nums[i]) > - 1) continue
       list.push(nums[i])
        dfs(list)
        list.pop()
    }
  }
  dfs([])
  return res
};
